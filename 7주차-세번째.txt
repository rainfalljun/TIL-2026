#include "listS.h"

int insertElement(int L[], int n, int x) {

    int i = 0, k = 0, move = 0;
    // i     = 반복문용 인덱스
    // k     = 삽입할 위치
    // move  = 자리이동 횟수 카운터

    for (i = 0; i < n; i++) {
        // 리스트를 앞에서부터 훑으면서 삽입 위치 탐색

        if (L[i] <= x && x <= L[i + 1]) {
            // x가 L[i]와 L[i+1] 사이에 들어갈 수 있으면
            // 예) 20 <= 30 <= 40 이면 20과 40 사이에 삽입

            k = i + 1;  // 삽입 위치는 i+1번째
            break;      // 위치 찾았으니 루프 종료
        }
    }

    if (i == n)
        // 루프를 끝까지 돌았다 = x가 가장 큰 원소보다 크다
        // 예) 리스트: 10 20 40 50 60 70 에서 x=80 이면
        k = n;  // 맨 뒤에 삽입

    for (i = n; i > k; i--) {
        // 마지막 원소부터 k+1까지 한 칸씩 뒤로 밀기
        // 뒤에서부터 미는 이유 : 앞에서 밀면 원소가 덮어써지기 때문
        // 예) k=2 일때: L[6]=L[5], L[5]=L[4], L[4]=L[3], L[3]=L[2]

        L[i] = L[i - 1];  // 한 칸 뒤로 이동
        move++;            // 이동 횟수 증가
    }

    L[k] = x;      // 비워진 k자리에 x 삽입
    return move;   // 이동 횟수 반환
}


int deleteElement(int L[], int n, int x) {

    int i, k = n, move = 0;
    // k = n 으로 초기화 : 원소를 못 찾았을 때를 대비한 기본값
    // 나중에 move==size 비교로 삭제 실패를 감지하는 데 사용

    for (i = 0; i < n; i++) {
        // 리스트를 앞에서부터 훑으면서 x 탐색

        if (L[i] == x) {
            // x를 찾으면
            k = i;   // 삭제할 위치 저장
            break;   // 찾았으니 루프 종료
        }
    }

    if (i == n)
        // 루프를 끝까지 돌았다 = x가 리스트에 없다
        move = n;
        // move를 n으로 설정 → main에서 삭제 실패 신호로 사용
        // 이때 k = n 이라서 아래 이동 루프도 실행되지 않음

    for (i = k; i < n - 1; i++) {
        // k 위치부터 앞으로 당기기
        // 앞에서부터 당기는 이유 : 뒤에서 당기면 원소가 덮어써지기 때문
        // 예) k=2 일때: L[2]=L[3], L[3]=L[4], L[4]=L[5], L[5]=L[6]

        L[i] = L[i + 1];  // 한 칸 앞으로 이동
        move++;            // 이동 횟수 증가
    }

    return move;  // 이동 횟수 반환 (실패시 n 반환)
}


---------- seqMain.c ----------

#include <stdio.h>
#include <windows.h>
#include "listS.h"

int main(void) {

    SetConsoleOutputCP(65001);  // 콘솔 한글 깨짐 방지 (출력)
    SetConsoleCP(65001);        // 콘솔 한글 깨짐 방지 (입력)

    int list[MAX] = { 10, 20, 40, 50, 60, 70 };
    // MAX=10 크기 배열, 앞 6칸만 사용
    // 나머지 4칸은 삽입을 위한 여유 공간

    int i = 0, move = 0, size = 6;
    // size = 현재 실제로 사용중인 원소 개수

    printf("\n삽입 전 선형 리스트 : ");
    for (i = 0; i < size; i++)    // 0~5번 인덱스 출력
        printf("%3d ", list[i]);  // %3d : 3칸 너비로 정렬해서 출력
    printf("\n원소의 갯수 : %d \n", size);  // 6 출력

    move = insertElement(list, size, 30);
    // 리스트에 30 삽입
    // 반환된 이동 횟수를 move에 저장

    printf("\n삽입 후 선형 리스트 : ");
    for (i = 0; i <= size; i++)  // 삽입 후라 size까지(포함) 출력
        printf("%3d ", list[i]);
    printf("\n원소의 갯수 : %d ", ++size);
    // ++size : size를 1 증가시키면서 출력 (6 → 7)
    printf("\n자리 이동 횟수 : %d \n", move);  // 4 출력

    printf("원소 30을 삭제전 size=%d\n", size);  // 7 출력

    move = deleteElement(list, size, 30);
    // 리스트에서 30 삭제
    // 반환된 이동 횟수를 move에 저장

    if (move == size)
        // move == size 이면 삭제 실패 (원소가 없었다는 신호)
        printf("\n선형 리스트에 원소가 없어서 삭제할 수 없습니다.\n");
    else {
        printf("\n삭제 후 선형 리스트 : ");
        for (i = 0; i < size - 1; i++)  // 삭제 후라 size-1개만 출력
            printf("%3d ", list[i]);
        printf("\n원소의 갯수 : %d ", --size);
        // --size : size를 1 감소시키면서 출력 (7 → 6)
        printf("\n자리 이동 횟수 : %d \n", move);  // 4 출력
    }

    return 0;
}

[2] 핵심 개념 요약
=====================================

-------------------------------------
이 코드의 목적
-------------------------------------
정렬된 순서(오름차순)를 유지하면서
숫자를 삽입하고 삭제하는 프로그램

초기 리스트: 10 20 40 50 60 70
30 삽입 →  10 20 30 40 50 60 70  (20과 40 사이에 자동으로 들어감)
30 삭제 →  10 20 40 50 60 70    (빈자리 없이 앞으로 당겨짐)


-------------------------------------
삽입 vs 삭제 비교
-------------------------------------
연산    이동방향    이동시작        이동횟수
삽입    뒤로 밀기   마지막부터      n - k
삭제    앞으로 당기기  k+1부터      n - k - 1


-------------------------------------
삽입 과정 (30 삽입, k=2)
-------------------------------------
전: [10][20][40][50][60][70][  ]
     0   1   2   3   4   5   6

뒤로 밀기 (4번):
L[6]=L[5] → L[5]=L[4] → L[4]=L[3] → L[3]=L[2]

후: [10][20][  ][40][50][60][70]
L[2] = 30 삽입

결과: [10][20][30][40][50][60][70]


-------------------------------------
삭제 과정 (30 삭제, k=2)
-------------------------------------
전: [10][20][30][40][50][60][70]
     0   1   2   3   4   5   6

앞으로 당기기 (4번):
L[2]=L[3] → L[3]=L[4] → L[4]=L[5] → L[5]=L[6]

결과: [10][20][40][50][60][70]


-------------------------------------
삭제 실패 감지 방법
-------------------------------------
deleteElement 함수에서
  - k = n 으로 초기화
  - 못 찾으면 move = n 으로 설정
  - k = n 이라 이동 루프 실행 안됨 (배열 변화 없음)

main에서
  if (move == size) → 삭제 실패!


-------------------------------------
자주 헷갈리는 포인트
-------------------------------------
1. 삽입할 때 출력 루프가 i <= size 인 이유
   → 삽입 후 원소가 1개 늘었기 때문

2. ++size vs size++
   ++size : 증가시키고 출력 (삽입 후 사용)
   size++ : 출력하고 증가 (이 코드에서 사용 안함)

3. %3d 를 쓰는 이유
   → 출력 너비를 3칸으로 고정해서 숫자를 정렬되게 출력

4. 삽입은 뒤에서부터 미는 이유
   → 앞에서 밀면 원소가 덮어써지기 때문

5. 삭제는 앞에서부터 당기는 이유
   → 뒤에서 당기면 원소가 덮어써지기 때문
