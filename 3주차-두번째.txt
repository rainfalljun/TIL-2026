알고리즘 수업 복습

#include <stdio.h> // 표준 입출력 함수를 사용하기 위한 헤더 파일 포함

// 리스트의 내용을 화면에 보여주는 함수 정의
void printList(int arr[], int size) {
    for (int i = 0; i < size; i++) { // 유효한 원소 개수만큼 반복
        printf("%d  ", arr[i]); // 배열의 원소를 출력
    }
    printf("\n원소의 갯수 : %d\n", size); // 현재 원소가 총 몇 개인지 출력
}

int main() {
    // 10칸짜리 배열을 만들고 초기값 저장
    int list[10] = { 10, 20, 40, 50, 60, 70, 0, 0, 0, 0 };
    int size = 6; // 현재 숫자가 들어있는 칸은 6개
    int moveCount = 0; // 데이터가 옆으로 이동한 횟수를 저장할 변수

    printf("Start 선형 리스트 원소 삽입과 삭제 연습\n");
    for (int i = 0; i < 10; i++) { // 전체 배열 10칸을 하나씩 확인
        printf("list[%d]=%d\n", i, list[i]); // 인덱스 번호와 저장된 값 출력
    }

    printf("\n삽입 전 선형 리스트 : ");
    printList(list, size); // 삽입 전 현재 상태 출력

    int insertIdx = 2; // 데이터를 넣고 싶은 위치(인덱스 2)
    int insertVal = 30; // 새로 넣을 숫자 30
    moveCount = 0; // 이동 횟수 초기화

    // 데이터를 삽입하기 위해 뒤에 있는 숫자들을 한 칸씩 뒤로 밀기
    for (int i = size; i > insertIdx; i--) { 
        list[i] = list[i - 1]; // 이전 칸의 값을 다음 칸으로 복사
        moveCount++; // 이동이 발생했으므로 횟수 증가
    }
    list[insertIdx] = insertVal; // 비워진 2번 자리에 숫자 30 저장
    size++; // 새로운 숫자가 들어왔으므로 전체 개수 1 증가

    printf("\n삽입 후 선형 리스트 : ");
    printList(list, size); // 삽입 결과 출력
    printf("자리 이동 횟수 : %d\n", moveCount); // 이동한 횟수 출력

    printf("원소 30을 삭제전 size=%d\n", size); // 삭제 전 현재 개수 확인
    int deleteIdx = 2; // 삭제하고 싶은 위치(인덱스 2)
    moveCount = 0; // 이동 횟수 초기화

    // 삭제된 빈 자리를 메우기 위해 뒤에 있는 숫자들을 한 칸씩 앞으로 당기기
    for (int i = deleteIdx; i < size - 1; i++) {
        list[i] = list[i + 1]; // 다음 칸의 값을 현재 칸으로 복사
        moveCount++; // 이동이 발생했으므로 횟수 증가
    }
    size--; // 숫자가 하나 빠졌으므로 전체 개수 1 감소
    list[size] = 0; // 맨 마지막에 남은 중복 데이터 자리를 0으로 정리

    printf("\n삭제 후 선형 리스트 : ");
    printList(list, size); // 삭제 결과 출력
    printf("자리 이동 횟수 : %d\n", moveCount); // 이동한 횟수 출력

    int userIdx, userVal; // 사용자가 입력할 값을 저장할 변수
    printf("\n원하는 배열의 인덱스와 원소값을 입력하시오 : ");
    scanf("%d %d", &userIdx, &userVal); // 키보드로부터 인덱스와 숫자 입력받기

    moveCount = 0; // 이동 횟수 초기화
    // 입력받은 위치에 숫자를 넣기 위해 다시 데이터들을 뒤로 밀기
    for (int i = size; i > userIdx; i--) {
        list[i] = list[i - 1]; // 한 칸씩 뒤로 복사
        moveCount++; // 이동 횟수 증가
    }
    list[userIdx] = userVal; // 사용자가 원한 위치에 숫자 저장
    size++; // 전체 개수 1 증가

    printf("\n삽입 후 선형 리스트 : ");
    printList(list, size); // 최종 결과 출력
    printf("자리 이동 횟수 : %d\n", moveCount); // 이동 횟수 출력

    return 0; // 프로그램 정상 종료
}
